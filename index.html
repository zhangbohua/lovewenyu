/**
*亲爱的夫人，认识你已经四年了；
*所以我想写一些东西来纪念或是感谢我们的这四年在一起的时光；
*/
Boy i = new Boy("张博华");
Girl u = new Girl("李文玉");
// May 2, 2014, 这是我人生中最重要的一天。这天，我第一次遇见了你.
// 从我看见你的第一眼，你对我笑的那一瞬间.
// 我觉得，我这辈子爱上你了。
i.love(u);
// Oct 13,2014,你答应我，并愿意成为我的女朋友.
u.accepted();
//从那天起，我每天每时每刻都在想着你，把你装在我心里。
i.miss(u);
//并且，我承诺用心照顾你一辈子，不让你伤心，不让你难过。
// 让你开开心心的，像个小公主。
i.takeCareOf(u);
// 我们也曾吵过小架，也曾为彼此伤心难过过，
// 但是经历过这四年的时光，我不说，你也懂；你不问，我也会去做。
// 只要两个人在一起，什么问题都可以解决。
// 是你让我懂得，有些东西是可以做出选择的。
boolean isWorked = true;
while (isWorked) {
u.waitFor(i);
//你毕业的时候，我还没有，没能陪你一起吃上班路上的早餐，喝下班路上的茶。
// 毕业之后我上班，一定要把你留在我身边，不让你离开我，一天也不可以.
isWorked = i.GraduateOver();
}
//几年之后,希望你能穿上洁白的婚纱，让我来娶你。
//夫人，余生还长，多多指教。
i.marry(u);
i.liveHappilyWith(u);
