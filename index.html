Boy i = new Boy("张博华");
Girl u = new Girl("李文玉");
// May 2, 2014, 我第一次遇见了你.
i.love(u);
// Oct 13,2014,你成为我的女朋友.
u.accepted();
//从那天起，我每天每时每刻都会想你 .
i.miss(u);
//并且，我承诺用心照顾你一辈子.
i.takeCareOf(u);
//即使你说过可能不喜欢我，不一定会和我走到最后.
// 但是，我还是用我所有的关心与呵护赢回了你的心,希望有一天能把你娶回家.
boolean isGraduate = true;
while (isGraduate) {
i.waitFor(u);
//你毕业的时候，我还没有，没能陪你一起上班，下班。
// 毕业之后，我一定要把你留在我身边，不让你离开我，一天也不可以.
isGraduate = i.GraduateOver();
}
//几年之后，一定为你举办一个最美的婚礼，让你成为最幸福的人。 老婆，我爱你。
i.marry(u);
i.liveHappilyWith(u);
